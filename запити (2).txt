#3)Показати імя  і телефон адміністраторів і директора салону

#4) Показати перших 10 клієнтів по алфафіну

#5)Показати клієнта чий номер закінчується на 15

#6)Показати імя номер телефону  працівників:  Oksana Gutsiv, Natalia Steslavska

    where FirstName like 'Oksana'and LastName like 'Gutsiv'

#7)показати ті послуги і ціни на них, які коштуюте дороще 500;

#8)Показати всі послуги які коштують дороще 600 і дешевше 900

#9) Порахувати яка загальна сума всіх послуг салону;

#10) Показати найдорощу послугу салону

#11) Показати всіх працівників які мають з/п більше 26000
select FirstName, Position  from workers join salary  s on workers.id = s.id
    where `size salary`>26000
        group by FirstName, Position;

#12) Показати 5 наймолодших працівників салону
select * from workers
    group by FirstName
        order by min(Age)
            limit 5;

#13) Показати інформацію про замовлення
select * from orders join clients c on orders.id = c.id
                    join services s on orders.`order` = s.id
group by orders.id;

#14) Показати інформацію про замовлення на суму більшу рівну 600
select * from orders join clients c on orders.id = c.id
                    join services s on orders.`order` = s.id
         where praise >= 600
            group by orders.id ;

#15) Показати всі послуги салону від найдорощї до найдешевшої
select * from services
    group by id
    order by min(praise) desc ;
